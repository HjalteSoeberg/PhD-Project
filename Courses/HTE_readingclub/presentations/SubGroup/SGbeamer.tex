\section{Introduction}
\label{sec:org591d8b5}

\begin{frame}[label={sec:org182dbca}]{Problem}
\begin{itemize}
\item Statins are great at treating Coronary artey disease, but may be associated with increased risk of T2D
\item Many studies have been conducted, but they either didnt have significant results (didnt use HTE) or were done as randomized controlled trials, which
has small sample size and might not be generelizable beyond the study population.
\end{itemize}
\end{frame}

\begin{frame}[label={sec:orgb8f69cc}]{Goal}
\begin{enumerate}
\item Designing a rigorous causal framework that systematically
examines the causal effects of statin usage on T2D risk from observational data.\pause
\item Uncovering which patient subgroup is most vulnerable for developing T2D after
taking statins.\pause
\item Assessing the replicability and statistical significance of the most vulnerable
subgroup via a bootstrap calibration procedure.
\end{enumerate}
\end{frame}

\begin{frame}[label={sec:org75767e8}]{Model Setup}
\begin{itemize}
\item Logistic Regression model, as outcome is binary.
\item Pre-defined subgroups instead of post-hoc indentified subgroups (better when we have
prior knowledge to avoid potential bias induced by dataadaptive subgroup identification).
\end{itemize}
\end{frame}

\begin{frame}[label={sec:orgbf09df0}]{Model Setup}
\begin{align*}
\text{logit}\{P(y = 1\vert z,x)\} = z^T\beta+x^T\gamma,\text{   } \Vert\gamma\Vert_0 \ll p
\end{align*}
\begin{itemize}
\item y is T2D status
\item \(z\in \mathbb{R}^{p_1\times n}\) are the subgroup indicator variables
\item \(x\in\mathbb{R}^{p_2\times n}\) is the intercept, the subgroup indicators and potential confounders
\item \(\beta\in\mathbb{R}^{p_1}\) represents the subgroup causal effects on the scale of log odds
ratio, i.e. \(\beta = \left(\text{log}\alpha_1,...,\log\alpha_{p_1}  \right)\) where
\(\log\alpha_j\) is the log odds ratio of subgroup \(j\).
\item It is assumed that \(\gamma\in
  \mathbb{R}^{p_2}\) is a sparse vector with support set \(M_0\).
\end{itemize}
\end{frame}

\begin{frame}[label={sec:org6a1a85e}]{Goal 1: Causal identification}
Show that the parameter of interest \(\beta\) represents subgroup
treatment effects under the model.\pause
\\[0pt]
This is done in the appendix but only for 2 and 6 subgroups.
\end{frame}

\begin{frame}[label={sec:orgc516480}]{Simulations}
Their simulations study is for \(p_1\in\{4,10\}\), while they only prove that \(\beta\)
represents subgroup treatment for \(p_1\in\{2,6\}\).
\end{frame}


\begin{frame}[label={sec:org4b9a807}]{Limitation}
\begin{itemize}
\item Only able to make inference on the most vulnerable subgroup.
\item Need pre-defined subgroups. Note that they argue that post-hoc identified groups can introduce
bias, so if we have prior knowledge it can be best to pre-define the subgroups.
\item Some proofs and examples are for their specific case and not generalized.
\end{itemize}
\end{frame}
